class Car{
    park(){
        return "stopped";
    }
    drive(){
        return "vroom";
    }
}

let car;
beforeEach(() =>{
    car=new Car();
})

describe ('car', () => {
    it('can park',() =>{
        assert.equal(car.park(),'stopped');
    }) 

    it('can drive',()=>{
        assert.equal(car.drive(),'vroom');
    })
})
